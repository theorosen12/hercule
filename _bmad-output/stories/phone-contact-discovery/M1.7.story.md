# Story M1.7: Module Header Actions

## Status: Done

## Story

**As a** user viewing the suggestion module,
**I want** to refresh suggestions or close the module permanently,
**so that** I have control over my experience.

## Acceptance Criteria

1. Create `ModuleHeader` component with title + action buttons
2. "Refresh" button (↻ icon):
   - Re-fetches contacts from device
   - Calls `getContactSuggestions(phoneNumbers)` directly (stateless - no upload)
   - Show loading state during refresh
   - Respect rate limits (show message if exceeded)
3. "Close" button (X icon):
   - Show confirmation dialog: "Ne plus afficher ce module ?"
   - On confirm: set `hide_friend_suggestion_profile` to `true` in AsyncStorage
   - Module immediately disappears
   - Module never reappears (until app data cleared)
4. Header title: "Amis trouvés dans vos contacts" (when suggestions) or "Trouvez vos amis" (when no phone)

## Technical Notes

- AsyncStorage key: `hide_friend_suggestion_profile`
- Confirmation dialog uses existing modal/alert pattern

## Implementation

```typescript
// ModuleHeader.component.tsx
import { Alert } from 'react-native';
import AsyncStorage from '@react-native-async-storage/async-storage';
import { useIntl } from 'react-intl';
import { moduleHeaderMessages } from './ModuleHeader.messages';
import * as S from './ModuleHeader.styles';

const HIDE_FLAG_KEY = 'hide_friend_suggestion_profile';

interface Props {
  title: string;
  onRefresh: () => void;
  onClose: () => void;
  isRefreshing: boolean;
}

export const ModuleHeader = ({ title, onRefresh, onClose, isRefreshing }: Props) => {
  const intl = useIntl();

  const handleClosePress = () => {
    Alert.alert(
      intl.formatMessage(moduleHeaderMessages.closeConfirmTitle),
      intl.formatMessage(moduleHeaderMessages.closeConfirmMessage),
      [
        {
          text: intl.formatMessage(moduleHeaderMessages.cancel),
          style: 'cancel',
        },
        {
          text: intl.formatMessage(moduleHeaderMessages.confirm),
          style: 'destructive',
          onPress: async () => {
            await AsyncStorage.setItem(HIDE_FLAG_KEY, 'true');
            onClose();
          },
        },
      ]
    );
  };

  return (
    <S.Container>
      <S.Title>{title}</S.Title>

      <S.ActionsContainer>
        <S.IconButton onPress={onRefresh} disabled={isRefreshing}>
          {isRefreshing ? (
            <S.Spinner />
          ) : (
            <RefreshIcon />
          )}
        </S.IconButton>

        <S.IconButton onPress={handleClosePress}>
          <CloseIcon />
        </S.IconButton>
      </S.ActionsContainer>
    </S.Container>
  );
};
```

## Refresh Logic (Stateless v3.1)

```typescript
// In FriendSuggestionModule.container.tsx
const handleRefresh = async () => {
  setIsRefreshing(true);

  try {
    // 1. Re-fetch contacts from device
    const { status } = await Contacts.getPermissionsAsync();

    if (status !== 'granted') {
      // Permission was revoked, hide module for session
      setSessionHidden(true);
      return;
    }

    const { data } = await Contacts.getContactsAsync({
      fields: [Contacts.Fields.PhoneNumbers],
    });

    const phoneNumbers = extractAndDedupePhones(data);

    // 2. Query suggestions directly (stateless - contacts passed as parameter, never stored)
    await getSuggestions({
      variables: { phoneNumbers },
      fetchPolicy: 'network-only', // Force fresh query
    });

  } catch (error) {
    const errors = extractRequestErrors(error);

    if (errors.RATE_LIMIT_EXCEEDED) {
      Alert.alert(intl.formatMessage(moduleHeaderMessages.rateLimitError));
    } else if (errors.CONTACT_LIMIT_EXCEEDED) {
      // Should not happen if we truncate, but handle gracefully
      Alert.alert(intl.formatMessage(moduleHeaderMessages.contactLimitError));
    } else {
      // Show generic error toast
      showErrorToast(error);
    }
  } finally {
    setIsRefreshing(false);
  }
};
```
