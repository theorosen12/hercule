# Story 3.5: Device Contacts Invite Section

## Status

Draft

## Story

**As a** new user on the Friend Discovery screen,
**I want** to see my device contacts who are not on Nara,
**So that** I can invite them to join the platform.

## Acceptance Criteria

1. Display device contacts in "Inviter des contacts" section
2. Initially show maximum 3 contacts
3. Show "Voir plus" button if more than 3 contacts exist
4. Each contact shows: name, phone number, "Inviter" button
5. "Inviter" button opens SMS with personalized invite message
6. Reuse invite logic from `NoSuggestionsState`

### InviteContactRow Component

```typescript
// components/InviteContactRow/InviteContactRow.component.tsx
import React from 'react';
import { useIntl } from 'react-intl';

import { Button } from '@nara/evanescent/components/Input/Button/Button.component';
import { Avatar } from '@nara/evanescent/components/Display/Avatar/Avatar.component';

import { useContactInvite } from '../../hooks/useContactInvite.hook';
import { DeviceContact } from '@domains/user/contactDiscovery/view/hooks/useContactDiscovery.hook';
import { messages } from './InviteContactRow.messages';
import * as S from './InviteContactRow.styles';

interface InviteContactRowProps {
  contact: DeviceContact;
  testID?: string;
}

export const InviteContactRow: React.FC<InviteContactRowProps> = ({
  contact,
  testID,
}) => {
  const intl = useIntl();
  const { inviteContact } = useContactInvite();

  const handleInvite = () => {
    inviteContact(contact);
  };

  // Format phone number for display (e.g., +33 6 12 34 56 78)
  const formatPhoneDisplay = (phone: string) => {
    // Simple formatting, could be enhanced
    return phone.replace(/(\+\d{2})(\d)(\d{2})(\d{2})(\d{2})(\d{2})/, '$1 $2 $3 $4 $5 $6');
  };

  return (
    <S.Container testID={testID}>
      <S.AvatarContainer>
        <Avatar
          size="medium"
          initials={contact.name?.charAt(0) || '?'}
        />
      </S.AvatarContainer>

      <S.InfoContainer>
        <S.NameText numberOfLines={1}>
          {contact.name || intl.formatMessage(messages.unknownContact)}
        </S.NameText>
        <S.PhoneText numberOfLines={1}>
          {formatPhoneDisplay(contact.phoneNumber)}
        </S.PhoneText>
      </S.InfoContainer>

      <S.ActionContainer>
        <Button
          label={intl.formatMessage(messages.inviteButton)}
          onPress={handleInvite}
          variant="secondary"
          size="small"
          testID={`${testID}-invite-button`}
        />
      </S.ActionContainer>
    </S.Container>
  );
};
```

### useContactInvite Hook

```typescript
// hooks/useContactInvite.hook.ts
import { useCallback } from 'react';
import { Linking, Platform } from 'react-native';
import { useIntl } from 'react-intl';

import { DeviceContact } from '@domains/user/contactDiscovery/view/hooks/useContactDiscovery.hook';
import { messages } from './useContactInvite.messages';

const INVITE_URL = __DEV__
  ? 'https://devlanding.nara-app.com'
  : 'https://nara-app.com';

export const useContactInvite = () => {
  const intl = useIntl();

  const inviteContact = useCallback(
    async (contact: DeviceContact) => {
      const firstName = contact.name?.split(' ')[0] || '';

      const message = intl.formatMessage(messages.smsInviteMessage, {
        name: firstName,
        url: INVITE_URL,
      });

      const phoneNumber = contact.phoneNumber.replace(/\s/g, '');

      // SMS URL format differs between iOS and Android
      const smsUrl = Platform.select({
        ios: `sms:${phoneNumber}&body=${encodeURIComponent(message)}`,
        android: `sms:${phoneNumber}?body=${encodeURIComponent(message)}`,
      });

      if (smsUrl) {
        await Linking.openURL(smsUrl);
      }
    },
    [intl],
  );

  return { inviteContact };
};
```
